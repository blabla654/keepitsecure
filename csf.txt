analyse email header - code:

File code.py:
    import email

    def parse_email_headers(filename):
        with open(filename, 'rb') as f:
            msg = email.message_from_binary_file(f)

        headers = {
            'From': msg['From'],
            'To': msg['To'],
            'Subject': msg['Subject'],
            'Date': msg['Date']
        }

        return headers

    if __name__ == '__main__':
        filename = 'email.txt'  # Replace with your email file name
        headers = parse_email_headers(filename)

        for header, value in headers.items():
            print(f"{header}: {value}")

Verify Captcha code:

File verifyCaptcha.py:
    import random
    import string

    def generate_captcha():
        letters = string.ascii_letters
        digits = string.digits
        captcha_list = [random.choice(letters + digits) for _ in range(6)]
        captcha_text = ''.join(captcha_list)
        return captcha_text

    def verify_captcha(captcha_text, user_input):
        if captcha_text == user_input:
            return True
        else:
            return False

    if __name__ == "__main__":
        captcha = generate_captcha()
        print("CAPTCHA:", captcha)

        user_input = input("Enter the CAPTCHA: ")

        if verify_captcha(captcha, user_input):
            print("CAPTCHA verified successfully!")
        else:
            print("CAPTCHA verification failed.")
